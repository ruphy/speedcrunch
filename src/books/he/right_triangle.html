A = (a * b) / 2

a = sqrt(c^2 - b^2)
b = sqrt(c^2 - a^2)
c = sqrt(a^2 + b^2)

a = sin(aa) * c
a = tan(aa) * b
a = cos(ba) * c
a = b / tan(ba)

b = sin(ba) * c
b = tan(ba) * a
b = cos(aa) * c
b = a / tan(aa)

c = a / sin(aa)
c = b / cos(aa)
c = b / sin(ba)
c = a / cos(ba)

aa = asin(a/c)
aa = acos(b/c)
